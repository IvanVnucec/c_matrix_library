546 void MTX_cholesky(MTX_Matrix_S *c, const MTX_Matrix_S *a, MTX_Error_E *error)
547 {
548  MTX_Error_E errorLocal = MTX_Matrix_ERROR_NONE;
549  int i, j, k;
550  float sum, val;
551 
552 #ifdef MTX_MATRIX_CHECK_PTRS
553  MTX_CHECK_NULL_PTRS_2(errorLocal, c, a);
554 #endif
555 
556 #ifdef MTX_MATRIX_CHECK_DIMS
557  if (errorLocal == MTX_Matrix_ERROR_NONE) {
558  MTX_MATRIX_CHECK_DIMS_2(errorLocal, a, c);
559  }
560 
561  if (errorLocal == MTX_Matrix_ERROR_NONE) {
562  // no need to check A matrix if its square because it is handled above
563  if (c->rows != c->cols) {
564  errorLocal = MTX_Matrix_ERROR_NOT_SQUARE;
565  }
566  }
567 #endif
568 
569  if (errorLocal == MTX_Matrix_ERROR_NONE) {
570  if (c->data == a->data) {
571  errorLocal = MTX_Matrix_ERROR_INPLACE;
572  }
573  }
574 
575  if (errorLocal == MTX_Matrix_ERROR_NONE) {
576  MTX_zeros(c, &errorLocal);
577  }
578 
579  if (errorLocal == MTX_Matrix_ERROR_NONE) {
580  for (i = 0; i < c->rows && errorLocal != MTX_Matrix_ERROR_NOT_POS_DEF; i++) {
581  for (j = 0; j < (i + 1) && errorLocal != MTX_Matrix_ERROR_NOT_POS_DEF; j++) {
582  sum = 0.0f;
583 
584  for (k = 0; k < j; k++) {
585  sum += c->data[i * c->cols + k] * c->data[j * c->cols + k];
586  }
587 
588  if (i == j) {
589  val = a->data[i * a->cols + i] - sum;
590 
591  if (val >= 0.0f) {
592  c->data[i * c->cols + j] = sqrtf(val);
593 
594  } else {
595  errorLocal = MTX_Matrix_ERROR_NOT_POS_DEF;
596  }
597 
598  } else {
599  c->data[i * c->cols + j] = 1.0f / c->data[j * c->cols + j]
600  * (a->data[i * a->cols + j] - sum);
601  }
602  }
603  }
604  }
